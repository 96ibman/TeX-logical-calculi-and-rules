\section{CNF}

% Disjunction Rules
\begin{multicols}{2}
    \noindent
    \begin{minipage}{\linewidth}
        \[
            \infer[\lor^T]
            {C \lor A^T \lor B^T}
            {C \lor (A \lor B)^T}
        \]
    \end{minipage}
    \noindent
    \begin{minipage}{\linewidth}
        \[
            \infer[\lor^F]
            {(C \lor A^F) \land (C \lor B^F)}
            {C \lor (A \lor B)^F}
        \]
    \end{minipage}
\end{multicols}

        \infer[\forall^T]
        {([Z/X](A))^T \lor C}
        {(\forall X.A)^T \lor C \quad Z \notin (free(A) \cup free(C))}
    \]
\end{minipage}
